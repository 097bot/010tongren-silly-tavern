<script>
(function(){
  let lastHostFont = null;

  function syncHostFontScale(force){
    try{
      const hostWin = (window.parent && window.parent !== window) ? window.parent : window;
      const hostDoc = hostWin.document;

      const candidates = [
        hostDoc.querySelector('.mes_text'),
        hostDoc.querySelector('.mes_reasoning'),
        hostDoc.querySelector('#send_textarea'),
        hostDoc.querySelector('#chat'),
        hostDoc.querySelector('.chat'),
        hostDoc.body,
        hostDoc.documentElement
      ].filter(Boolean);

      let hostSize = 16;

      for(const el of candidates){
        const size = parseFloat(hostWin.getComputedStyle(el).fontSize);
        if(size && !Number.isNaN(size)){
          hostSize = size;
          break;
        }
      }

      if(force || lastHostFont === null || Math.abs(hostSize - lastHostFont) > 0.05){
        lastHostFont = hostSize;
        document.documentElement.style.fontSize = hostSize + 'px';
      }
    }catch(e){
      if(force && !document.documentElement.style.fontSize){
        document.documentElement.style.fontSize = '16px';
      }
    }
  }

  function dec(s){
    return s
      .replace(/&amp;/g,'&')
      .replace(/&lt;/g,'<')
      .replace(/&gt;/g,'>')
      .replace(/&nbsp;/g,' ')
      .replace(/&quot;/g,'"')
      .replace(/→/g,'→')
      .replace(/≥/g,'≥')
      .replace(/≤/g,'≤');
  }

  function get(t,k){
    var m=t.match(new RegExp(k+'[\\uff1a:][^\\S\\n]*([^\\n]+)'));
    return m?dec(m[1].trim()):'';
  }

  syncHostFontScale(true);
  window.addEventListener('resize', syncHostFontScale);
  setInterval(syncHostFontScale, 600);

  var raw=dec(document.getElementById('gc-raw').innerHTML);

  document.getElementById('gc-title').textContent=get(raw,'标题');
  document.getElementById('gc-desc').textContent=get(raw,'说明');

  var container=document.getElementById('gc-options');
  var labels=['A','B','C','D','E','F'];

  for(var i=1;i<=6;i++){
    var val=get(raw,'选项'+i);
    if(!val) continue;

    var opt=document.createElement('div');
    opt.className='gc-option';
    opt.setAttribute('data-action',val);
    opt.innerHTML='<span class="gc-opt-label">'+labels[i-1]+'</span><span class="gc-opt-text">'+val+'</span>';

    opt.addEventListener('click',(function(v){
      return function(e){
        e.stopPropagation();
        try{
          var t=window.parent.document.querySelector('#send_textarea');
          var slash=window.parent.triggerSlash;

          if(t){
            var nv=t.value+v;
            if(slash){
              slash('/setinput '+nv);
            }else{
              t.value=nv;
              t.dispatchEvent(new Event('input',{bubbles:true}));
            }
          }else if(slash){
            slash('/setinput '+v);
          }
        }catch(e){}

        this.style.borderColor='rgba(200,168,74,0.8)';
        this.style.background='rgba(138,122,80,0.2)';
        var self=this;
        setTimeout(function(){
          self.style.borderColor='';
          self.style.background='';
        },600);
      };
    })(val));

    container.appendChild(opt);
  }

  var risk=get(raw,'风险提示');
  var check=get(raw,'推荐检定');
  document.getElementById('gc-risk').textContent=risk||'—';
  document.getElementById('gc-check').textContent=check||'无';

  if(!risk&&!check){
    document.getElementById('gc-footer').style.display='none';
  }

  document.getElementById('gc-header').addEventListener('click',function(){
    document.getElementById('gc-wrap').classList.toggle('collapsed');
  });
})();
</script>
