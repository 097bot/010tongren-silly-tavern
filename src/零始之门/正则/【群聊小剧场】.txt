  <script>
    (function(){
      var scriptEl = document.currentScript;
      var root = scriptEl.previousElementSibling;
      if(!root || !root.classList.contains('chat-root')) return;

      var wrap = root.querySelector('.chat-wrap');
      var header = root.querySelector('.chat-header');
      var titleEl = root.querySelector('.chat-title');
      var timeEl = root.querySelector('.chat-time');
      var timenodeEl = root.querySelector('.chat-timenode');
      var messagesEl = root.querySelector('.chat-messages');
      var rawEl = root.querySelector('.chat-raw');

      if(!wrap || !header || !titleEl || !timeEl || !timenodeEl || !messagesEl || !rawEl) return;

      let lastHostFont = null;

      function syncHostFontScale(force){
        try{
          const hostWin = (window.parent && window.parent !== window) ? window.parent : window;
          const hostDoc = hostWin.document;

          const candidates = [
            hostDoc.querySelector('.mes_text'),
            hostDoc.querySelector('.mes_reasoning'),
            hostDoc.querySelector('#send_textarea'),
            hostDoc.querySelector('#chat'),
            hostDoc.querySelector('.chat'),
            hostDoc.body,
            hostDoc.documentElement
          ].filter(Boolean);

          let hostSize = 16;

          for(const el of candidates){
            const size = parseFloat(hostWin.getComputedStyle(el).fontSize);
            if(size && !Number.isNaN(size)){
              hostSize = size;
              break;
            }
          }

          if(force || lastHostFont === null || Math.abs(hostSize - lastHostFont) > 0.05){
            lastHostFont = hostSize;
            document.documentElement.style.fontSize = hostSize + 'px';
          }
        }catch(e){
          if(force && !document.documentElement.style.fontSize){
            document.documentElement.style.fontSize = '16px';
          }
        }
      }

      function dec(s){
        return (s || '')
          .replace(/&amp;/g,'&')
          .replace(/&lt;/g,'<')
          .replace(/&gt;/g,'>')
          .replace(/&nbsp;/g,' ')
          .replace(/&quot;/g,'"');
      }

      function escapeHtml(s){
        return (s || '')
          .replace(/&/g,'&amp;')
          .replace(/</g,'&lt;')
          .replace(/>/g,'&gt;')
          .replace(/"/g,'&quot;');
      }

      function get(text, key){
        var m = text.match(new RegExp(key + '[：:][^\\S\\n]*([^\\n]+)'));
        return m ? dec(m[1].trim()) : '';
      }

      function hexToRgba(hex,a){
        var r=parseInt(hex.slice(1,3),16),
            g=parseInt(hex.slice(3,5),16),
            b=parseInt(hex.slice(5,7),16);
        return 'rgba('+r+','+g+','+b+','+a+')';
      }

      function getAvatarInitial(sender){
        if(!sender) return '?';
        var name = sender.trim();
        if(!name) return '?';
        if(/\s/.test(name)){
          return name.split(/\s+/)[0].slice(0,1).toUpperCase();
        }
        return name.slice(0,1);
      }

      function normType(s){
        s = dec(s || '').trim().toLowerCase();

        if(/^(sys|系统提示|系统|提示|system)$/.test(s)) return 'sys';
        if(/^(recall|撤回|撤回消息|撤回提示)$/.test(s)) return 'recall';
        if(/^(right|右|右侧|右边|发送者|本人|我方|自己)$/.test(s)) return 'right';
        if(/^(left|左|左侧|左边|对方)$/.test(s)) return 'left';

        return '';
      }

      function isLikelyRecallText(s){
        s = dec(s || '').trim();
        return /撤回了一条消息|撤回消息|已撤回|消息已撤回/.test(s);
      }

      function isLikelySystemText(s){
        s = dec(s || '').trim();
        return /(邀请.*加入群聊|邀请.*加入频道|加入了?.*群聊|加入了?.*频道|创建了?.*群聊|创建了?.*频道|修改了?.*群名|修改了?.*频道名称|拍了拍|被移出群聊|移出了群聊|离开了?.*群聊|离开了?.*频道|进入了?.*频道|通过了朋友验证|以上是打招呼内容|现在可以开始聊天|群公告|更换群头像|开启了群禁言|关闭了群禁言)/.test(s);
      }

      function isLikelyUserName(s){
        s = dec(s || '').trim();
        if(!s) return false;
        if(/[：:]/.test(s)) return false;
        if(s.length > 12) return false;
        if(/^(sys|系统提示|系统|提示|recall|撤回|left|right|左|右)$/i.test(s)) return false;
        return true;
      }

      function inferSideFromSender(sender){
        sender = dec(sender || '').trim();
        if(!sender) return 'left';
        if(/^\{\{user\}\}$|^你$|^我$|^自己$|^玩家$|^user$/i.test(sender)) return 'right';
        return 'left';
      }

      function extractPayload(line){
        line = dec(line || '').trim();
        if(!line) return '';

        if(/^<\/?群聊小剧场>/i.test(line)) return '';
        if(/^(群名|时间|时间节点)[：:]/.test(line)) return '';

        var m = line.match(/^(?:消息|msg|message)\s*\d*\s*[：:]\s*(.+)$/i);
        if(m) return m[1].trim();

        m = line.match(/^\d+\s*[.)、．]\s*(.+)$/);
        if(m) return m[1].trim();

        m = line.match(/^[\-•·]\s*(.+)$/);
        if(m) line = m[1].trim();

        if(line.indexOf('|') >= 0) return line;
        if(isLikelyRecallText(line)) return line;
        if(isLikelySystemText(line)) return line;

        var colonMsg = line.match(/^([^：:]{1,12})[：:]\s*(.+)$/);
        if(colonMsg && isLikelyUserName(colonMsg[1])) return line;

        return '';
      }

      function parseMessagePayload(payload){
        var text = dec(payload || '').trim();
        var result = {
          type: '',
          sender: '',
          content: '',
          quote: ''
        };

        if(!text) return result;

        var colonMsg = text.match(/^([^：:]{1,12})[：:]\s*(.+)$/);
        if(colonMsg && isLikelyUserName(colonMsg[1]) && !normType(colonMsg[1])){
          result.sender = colonMsg[1].trim();
          result.content = colonMsg[2].trim();
          result.type = inferSideFromSender(result.sender);
          return result;
        }

        var parts = text.split(/\s*\|\s*/).map(function(x){
          return dec(x.trim());
        }).filter(function(x){ return x !== ''; });

        if(parts.length === 0) return result;

        var t0 = normType(parts[0]);
        var t1 = parts[1] ? normType(parts[1]) : '';

        if(t0){
          result.type = t0;

          if(result.type === 'sys'){
            result.content = parts.slice(1).join('|').trim();
            return result;
          }

          if(result.type === 'recall'){
            if(parts.length >= 3){
              result.sender = (parts[1] || '').trim();
              result.content = parts.slice(2).join('|').trim();
            }else{
              result.content = parts.slice(1).join('|').trim();
            }
            return result;
          }

          if(result.type === 'right'){
            if(parts.length >= 3){
              result.sender = (parts[1] || '').trim() || '{{user}}';
              result.content = (parts[2] || '').trim();
              result.quote = parts.length > 3 ? parts.slice(3).join('|').trim() : '';
            }else{
              result.sender = '{{user}}';
              result.content = parts.slice(1).join('|').trim();
            }
            return result;
          }

          if(result.type === 'left'){
            if(parts.length >= 3){
              result.sender = (parts[1] || '').trim() || '未知';
              result.content = (parts[2] || '').trim();
              result.quote = parts.length > 3 ? parts.slice(3).join('|').trim() : '';
            }else if(parts.length === 2){
              result.sender = '未知';
              result.content = parts[1].trim();
            }
            return result;
          }
        }

        if(t1){
          result.type = t1;

          if(result.type === 'sys'){
            result.content = [parts[0]].concat(parts.slice(2)).join('|').trim();
            return result;
          }

          if(result.type === 'recall'){
            result.sender = (parts[0] || '').trim();
            result.content = parts.slice(2).join('|').trim();
            return result;
          }

          result.sender = (parts[0] || '').trim() || (result.type === 'right' ? '{{user}}' : '未知');
          result.content = (parts[2] || '').trim();
          result.quote = parts.length > 3 ? parts.slice(3).join('|').trim() : '';
          return result;
        }

        if(parts.length === 1){
          if(isLikelyRecallText(parts[0])){
            result.type = 'recall';
            result.content = parts[0];
            return result;
          }
          if(isLikelySystemText(parts[0])){
            result.type = 'sys';
            result.content = parts[0];
            return result;
          }
          result.type = 'sys';
          result.content = parts[0];
          return result;
        }

        if(parts.length === 2){
          if(isLikelyRecallText(text)){
            result.type = 'recall';
            result.sender = isLikelyUserName(parts[0]) ? parts[0] : '';
            result.content = parts[1] || text;
            return result;
          }

          if(isLikelySystemText(text)){
            result.type = 'sys';
            result.content = text;
            return result;
          }

          if(isLikelyUserName(parts[0])){
            result.sender = parts[0];
            result.content = parts[1];
            result.type = inferSideFromSender(result.sender);
            return result;
          }

          result.type = 'sys';
          result.content = text;
          return result;
        }

        if(parts.length >= 3){
          if(isLikelyUserName(parts[0])){
            result.sender = parts[0];
            result.content = parts[1];
            result.quote = parts.slice(2).join('|').trim();
            result.type = inferSideFromSender(result.sender);
            return result;
          }

          result.type = 'sys';
          result.content = text;
          return result;
        }

        result.type = 'sys';
        result.content = text;
        return result;
      }

      syncHostFontScale(true);
      window.addEventListener('resize', syncHostFontScale);
      setInterval(syncHostFontScale, 800);

      var raw = dec(rawEl.innerHTML);

      var groupName = get(raw,'群名');
      var time = get(raw,'时间');
      var timenode = get(raw,'时间节点');

      titleEl.textContent = '▌ ' + (groupName || '群聊');

      timenodeEl.textContent = timenode || '';
      if(!timenode) timenodeEl.style.display = 'none';

      var avatarColors = {};
      var colorPool = ['#4a6741','#c8a84a','#7a5cbf','#3a8fc8','#c86060','#c87a40','#508060'];
      var colorIdx = 0;

      function getColor(name){
        if(!avatarColors[name]){
          avatarColors[name] = colorPool[colorIdx % colorPool.length];
          colorIdx++;
        }
        return avatarColors[name];
      }

      var lines = raw.split('\n');

      for(var i=0;i<lines.length;i++){
        var payload = extractPayload(lines[i]);
        if(!payload) continue;

        var msg = parseMessagePayload(payload);
        if(!msg.type) continue;

        var el = document.createElement('div');

        if(msg.type === 'sys'){
          el.className = 'chat-sys';
          el.innerHTML = '<span class="chat-sys-pill">' + escapeHtml(msg.content || '系统提示') + '</span>';
        }
        else if(msg.type === 'recall'){
          var recallText = '';
          if(msg.sender && msg.content){
            recallText = msg.sender + (/\s/.test(msg.content) ? '' : ' ') + msg.content;
          }else{
            recallText = msg.content || msg.sender || '撤回了一条消息';
          }
          el.className = 'chat-recall';
          el.innerHTML = '<span class="chat-recall-pill">' + escapeHtml(recallText) + '</span>';
        }
        else if(msg.type === 'right'){
          var qh = msg.quote ? '<div class="chat-quote">' + escapeHtml(msg.quote) + '</div>' : '';
          el.className = 'chat-row-right';
          el.innerHTML = '<div class="chat-bubble-right">' + qh + escapeHtml(msg.content || '') + '</div>';
        }
        else{
          var sender = (msg.sender || '未知');
          var content = (msg.content || '');
          var quote = (msg.quote || '');
          var col = getColor(sender);
          var init = getAvatarInitial(sender);
          var qh = quote ? '<div class="chat-quote">' + escapeHtml(quote) + '</div>' : '';
          el.className = 'chat-row-left';
          el.innerHTML =
            '<div class="chat-avatar" style="background:' + hexToRgba(col,0.3) + ';border:1px solid ' + hexToRgba(col,0.5) + ';">' + escapeHtml(init) + '</div>' +
            '<div class="chat-col">' +
              '<div class="chat-sender">' + escapeHtml(sender) + '</div>' +
              '<div class="chat-bubble-left">' + qh + escapeHtml(content) + '</div>' +
            '</div>';
        }

        messagesEl.appendChild(el);
      }

      header.addEventListener('click', function(e){
        e.preventDefault();
        e.stopPropagation();
        wrap.classList.toggle('collapsed');
      });
    })();
  </script>
