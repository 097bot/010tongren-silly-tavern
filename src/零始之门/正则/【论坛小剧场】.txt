  <script>
    (function(){
      let lastHostFont = null;

      function syncHostFontScale(force){
        try{
          const hostWin = (window.parent && window.parent !== window) ? window.parent : window;
          const hostDoc = hostWin.document;

          const candidates = [
            hostDoc.querySelector('.mes_text'),
            hostDoc.querySelector('.mes_reasoning'),
            hostDoc.querySelector('#send_textarea'),
            hostDoc.querySelector('#chat'),
            hostDoc.querySelector('.chat'),
            hostDoc.body,
            hostDoc.documentElement
          ].filter(Boolean);

          let hostSize = 16;

          for(const el of candidates){
            const size = parseFloat(hostWin.getComputedStyle(el).fontSize);
            if(size && !Number.isNaN(size)){
              hostSize = size;
              break;
            }
          }

          if(force || lastHostFont === null || Math.abs(hostSize - lastHostFont) > 0.05){
            lastHostFont = hostSize;
            document.documentElement.style.fontSize = hostSize + 'px';
          }
        }catch(e){
          if(force && !document.documentElement.style.fontSize){
            document.documentElement.style.fontSize = '16px';
          }
        }
      }

      function dec(s){
        return (s || '')
          .replace(/&amp;/g,'&')
          .replace(/&lt;/g,'<')
          .replace(/&gt;/g,'>')
          .replace(/&nbsp;/g,' ')
          .replace(/&quot;/g,'"')
          .replace(/→/g,'→')
          .replace(/【/g,'【')
          .replace(/】/g,'】');
      }

      function get(t,k){
        var m=t.match(new RegExp(k+'[：:][^\\S\\n]*([^\\n]+)'));
        return m ? dec(m[1].trim()) : '';
      }

      function isValidHex(hex){
        return /^#([0-9a-fA-F]{6})$/.test(hex || '');
      }

      function hexToRgba(hex,a){
        if(!isValidHex(hex)) hex='#c8a84a';
        var r=parseInt(hex.slice(1,3),16),
            g=parseInt(hex.slice(3,5),16),
            b=parseInt(hex.slice(5,7),16);
        return 'rgba('+r+','+g+','+b+','+a+')';
      }

      function escapeHtml(s){
        return (s || '')
          .replace(/&/g,'&amp;')
          .replace(/</g,'&lt;')
          .replace(/>/g,'&gt;')
          .replace(/"/g,'&quot;');
      }

      function linesOf(text){
        return (text || '')
          .split(/\n+/)
          .map(function(x){ return x.trim(); })
          .filter(Boolean);
      }

      syncHostFontScale(true);
      window.addEventListener('resize', syncHostFontScale);
      setInterval(syncHostFontScale, 600);

      var raw = dec(document.getElementById('f-raw').innerHTML);

      var platform = get(raw,'平台');
      var section  = get(raw,'板块');
      var accent   = get(raw,'主色调') || '#c8a84a';
      var title    = get(raw,'标题');
      var info     = get(raw,'元信息');

      var wrap = document.getElementById('forum-wrap');
      wrap.style.setProperty('--accent',hexToRgba(accent,0.6));
      wrap.style.borderColor = hexToRgba(accent,0.4);

      var header = document.getElementById('forum-header');
      header.style.background = hexToRgba(accent,0.18);
      header.style.borderBottomColor = hexToRgba(accent,0.3);

      document.getElementById('fh-platform').textContent = '▌ ' + (platform || '平台');
      document.getElementById('fh-section').textContent = section || '';
      document.getElementById('f-section').textContent = section || '';

      var badge = document.getElementById('f-badge');
      badge.textContent = platform || '论坛';
      badge.style.background = hexToRgba(accent,0.2);
      badge.style.color = isValidHex(accent) ? accent : '#c8a84a';
      badge.style.border = '1px solid ' + hexToRgba(accent,0.45);

      document.getElementById('f-title').textContent = title || '';
      document.getElementById('f-info').textContent = info || '';

      var bodyEl = document.getElementById('f-body');
      var container = document.getElementById('f-comments');
      var colors = ['#e6162d','#4a6741','#c8a84a','#7a5cbf','#3a8fc8','#c86060','#5a8f7b','#b86aa8'];

      var comments = [];
      var bodyText = '';

      var bodyMatch = raw.match(/正文[：:][ \t]*([^\n]*)(?:\n([\s\S]*?))?(?=\n(?:用户\d*[（(]|评论\d*[（(]|点赞\d*[（(]|用户\d+[：:]|评论\d+[：:]|点赞\d+[：:]|用户[（(]|评论[（(]|点赞[（(]|<\/论坛小剧场>|$)|$)/);
      if(bodyMatch){
        var firstLine = dec((bodyMatch[1] || '').trim());
        var restBlock = dec((bodyMatch[2] || '').trim());
        bodyText = [firstLine, restBlock].filter(Boolean).join('\n').trim();
      }

      for(var i=1;i<=20;i++){
        var user = get(raw,'用户'+i);
        var comment = get(raw,'评论'+i);
        var like = get(raw,'点赞'+i);
        if(!user || !comment) break;
        comments.push({
          user:user,
          comment:comment,
          like:like
        });
      }

      if(comments.length === 0){
        var blockAfterBody = raw.match(/正文[：:][^\n]*\n?([\s\S]*)$/);
        var candidateText = blockAfterBody ? dec(blockAfterBody[1].trim()) : raw;

        var parsedComments2 = [];
        var pureBody2 = [];

        linesOf(candidateText).forEach(function(line){
          var speak = line.match(/^(用户|评论)\s*(\d+)?(?:\s*[（(]\s*([^）)]+)\s*[）)])?\s*[：:]\s*(.+)$/);
          var like  = line.match(/^点赞\s*(\d+)?(?:\s*[（(]\s*([^）)]+)\s*[）)])?\s*[：:]\s*(.+)$/);

          if(speak){
            var label = dec((speak[1] || '').trim());
            var idx   = dec((speak[2] || '').trim());
            var name  = dec((speak[3] || '').trim());
            var text  = dec((speak[4] || '').trim());

            parsedComments2.push({
              user: name || (label + (idx || '')),
              comment: text,
              like: '',
              _idx: idx || '',
              _name: name || ''
            });
          }else if(like){
            var likeIdx   = dec((like[1] || '').trim());
            var likeName  = dec((like[2] || '').trim());
            var likeText  = dec((like[3] || '').trim());

            var attached = false;

            for(var j=parsedComments2.length-1; j>=0; j--){
              var item = parsedComments2[j];

              if(likeIdx && likeName){
                if(item._idx === likeIdx && item._name === likeName){
                  item.like = likeText;
                  attached = true;
                  break;
                }
              }else if(likeIdx){
                if(item._idx === likeIdx){
                  item.like = likeText;
                  attached = true;
                  break;
                }
              }else if(likeName){
                if(item._name === likeName || item.user === likeName){
                  item.like = likeText;
                  attached = true;
                  break;
                }
              }
            }

            if(!attached && parsedComments2.length > 0){
              parsedComments2[parsedComments2.length - 1].like = likeText;
            }
          }else{
            pureBody2.push(line);
          }
        });

        if(parsedComments2.length > 0){
          comments = parsedComments2.map(function(item){
            return {
              user: item.user,
              comment: item.comment,
              like: item.like || ''
            };
          });

          if(pureBody2.length > 0){
            bodyText = pureBody2.join('\n');
          }
        }
      }

      if(comments.length === 0 && bodyText){
        var parsedComments3 = [];
        var pureBody3 = [];

        linesOf(bodyText).forEach(function(line){
          var m = line.match(/^([^：:\n]{1,30})[：:]\s*(.+)$/);
          if(m){
            parsedComments3.push({
              user:dec(m[1].trim()),
              comment:dec(m[2].trim()),
              like:''
            });
          }else{
            pureBody3.push(line);
          }
        });

        if(parsedComments3.length > 0){
          comments = parsedComments3;
          bodyText = pureBody3.join('\n');
        }
      }

      bodyEl.textContent = bodyText || '';

      comments.forEach(function(item,idx){
        var col = colors[idx % colors.length];
        var div = document.createElement('div');
        div.className = 'forum-comment';
        div.innerHTML =
          '<div class="forum-avatar" style="background:'+col+';opacity:0.75;"></div>' +
          '<div class="forum-comment-right">' +
            '<div class="forum-username">'+escapeHtml(item.user)+'</div>' +
            '<div class="forum-comment-text">'+escapeHtml(item.comment)+'</div>' +
            (item.like ? '<div class="forum-like">❤️ '+escapeHtml(item.like)+'</div>' : '') +
          '</div>';
        container.appendChild(div);
      });

      document.getElementById('forum-header').addEventListener('click',function(){
        document.getElementById('forum-wrap').classList.toggle('collapsed');
      });
    })();
  </script>
</body>
