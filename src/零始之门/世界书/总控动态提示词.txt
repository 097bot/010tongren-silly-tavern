function ctrl_entriesOf(obj){
  if (!obj || typeof obj !== 'object') return [];
  return Object.entries(obj).filter(([_, v]) => v && typeof v === 'object');
}

function ctrl_realCrises(obj){
  return ctrl_entriesOf(obj).filter(([_, v]) => {
    const target = String(v['对象'] || '');
    const note = String(v['备注'] || '');
    return target && target !== '无' && !/初始占位|无实际危机/.test(note);
  });
}

function ctrl_hasKretMonster(obj){
  return ctrl_entriesOf(obj).some(([_, v]) => /克锐特/.test(String(v['所属'] || '')));
}

function ctrl_hasPlan(obj){
  return ctrl_entriesOf(obj).some(([_, p]) =>
    ['待完成', '进行中', '搁置'].includes(String(p['状态'] || '待完成'))
  );
}

const place = String(world['场所'] || '');

const currentCP = String(relation['当前主CP'] || '');
const relationEvent = String(relation['当前关系事件'] || '无');

const round = Number(battle['回合'] || 0);

const hasEnemies = enemyList.length > 0;

const hasRealCrisis = realCrises.length > 0;

const risky = !!getvar('stat_data.世界.高风险判定状态');

const hasMonsterFocus = ctrl_hasKretMonster(monsters);
const hasPlanFocus = ctrl_hasPlan(plans);

const bagChange = String(bag['本轮变化'] || '无变化');
const hasBagFocus = bagChange !== '无变化';

let mainFocus = '日常推进';

else if (inBattle || round > 0 || hasEnemies) mainFocus = '战斗';
else if (hasJudgeFocus) mainFocus = '判定后果';
else if (hasMonsterFocus) mainFocus = '怪物与掉落';
else if (hasBagFocus) mainFocus = '资源与背包';
else if (currentCP || relationEvent !== '无') mainFocus = '关系';
else if (eventName !== '无' || place) mainFocus = '区域与遭遇';
else if (hasPlanFocus) mainFocus = '计划推进';

function ctrl_priorityList(){
  const list = [];
  if (hasRealCrisis) list.push('危机');
  if (inBattle || round > 0 || hasEnemies) list.push('战斗');
  if (hasJudgeFocus) list.push('判定后果');
  if (hasMonsterFocus) list.push('怪物与掉落');
  if (hasBagFocus) list.push('资源与背包');
  if (currentCP || relationEvent !== '无') list.push('关系');
  if (eventName !== '无' || place) list.push('区域与遭遇');
  if (hasPlanFocus) list.push('计划推进');
  if (list.length === 0) list.push('日常推进');
  return list;
}

const priorities = ctrl_priorityList();
